@Test
    public void npe(){
        Integer cycleIdStart = 202011;
        Integer cycleIdEnd  = 202101;

        LocalDate start = getCycleIdDate(cycleIdStart,2);
        LocalDate end = getCycleIdDate(cycleIdEnd,0);
        while (!start.isEqual(end)){
            start = start.plusMonths(1);
        }
    }
    private LocalDate getCycleIdDate(Integer cycleId,int plusMonth){
        Objects.requireNonNull(cycleId);
        return LocalDate.of(getYear(cycleId),getMonth(cycleId),1).plusMonths(plusMonth);
    }
    private static int getYear(Integer cycleId){
        return Integer.valueOf(cycleId.toString().substring(0,4));
    }
    private static int getMonth(Integer cycleId){
        return Integer.valueOf(cycleId.toString().substring(4,6));
    }
